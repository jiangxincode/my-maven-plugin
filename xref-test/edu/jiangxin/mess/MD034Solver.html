4   import java.io.BufferedWriter;

8   import java.io.IOException;
9   import java.util.regex.Matcher;
10  import java.util.regex.Pattern;
11  
12  /**
13   * Solve the problem of MD034. <p>
14   * {@link https://github.com/markdownlint/markdownlint/blob/master/docs/RULES.md}
15   * This program doesn't process any exception!
16   * @author aloys
17   *
18   */
19  public class MD034Solver {
20  	public static final String regex = "\\b(https?|ftp|file)://[-A-Z0-9+&@#/%?=~_|$!:,.;]*[A-Z0-9+&@#/%=~_|$]";
21  	
22  	public static final String sourceDirStr = "D:\\temp\\cnblogs";
23  	
24  	public static final String targetDirStr = "D:\\temp\\cnblogsbak";
25  
26  	public static void main(String[] args) throws IOException {
27  		// Source directory must be a valid directory which contains the text files to be processed.
28  		File sourceDir = new File(sourceDirStr);
29  
30  		// Target directory must be a valid directory which will save the proecessed files.
31  		File targetDir = new File(targetDirStr);
32  
33  		for (File file : sourceDir.listFiles()) {
34  
35  			// take off the .git directory and .gitignore file
36  			if (file.getName().startsWith(".")) {
37  				continue;
38  			}
39  
40  			BufferedReader reader = new BufferedReader(new FileReader(file));
41  			BufferedWriter writer = new BufferedWriter(new FileWriter(new File(targetDir, file.getName())));
42  
43  			String temp = null;
44  			while ((temp = reader.readLine()) != null) {
45  				Pattern pattern = Pattern.compile(regex, Pattern.CASE_INSENSITIVE | Pattern.UNICODE_CASE);
46  				Matcher regexMatcher = pattern.matcher(temp);
47  				StringBuffer sb = new StringBuffer();
48  				while (regexMatcher.find()) {
49  					String group = regexMatcher.group();
50  					int start = regexMatcher.start();
51  					if (start >= 1 && (temp.charAt(start - 1) == '<' || temp.charAt(start - 1) == '(')) {
52  						regexMatcher.appendReplacement(sb, group);
53  					} else {
54  						regexMatcher.appendReplacement(sb, "<" + group + ">");
55  					}
56  
57  				}
58  				regexMatcher.appendTail(sb);
59  				writer.write(sb.toString());
60  				writer.newLine();
61  			}
62  			reader.close();
63  			writer.close();
64  		}
65  	}
66  
67  }
