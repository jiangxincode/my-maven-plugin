15  public class ProxySettingUtils {
16  
17  	public static void global() throws IOException {
18  		Properties fileProperties = new Properties();
19  
20  		FileInputStream fis = new FileInputStream(new File("proxy.properties"));
21  		fileProperties.load(fis);
22  		fis.close();
23  
24  		Properties sysProperties = System.getProperties();
25  
26  		// 设置http访问要使用的代理服务器的地址
27  		sysProperties.setProperty("http.proxyHost", fileProperties.getProperty("http.proxyHost"));
28  		// 设置http访问要使用的代理服务器的端口
29  		sysProperties.setProperty("http.proxyPort", "8080");
30  		// 设置不需要通过代理服务器访问的主机，可以使用*通配符，多个地址用|分隔
31  		sysProperties.setProperty("http.nonProxyHosts", "localhost|192.168.0.*");
32  
33  		// 设置安全访问使用的代理服务器地址与端口
34  		// 它没有https.nonProxyHosts属性，它按照http.nonProxyHosts 中设置的规则访问
35  		sysProperties.setProperty("https.proxyHost", "183.45.78.31");
36  		sysProperties.setProperty("https.proxyPort", "443");
37  
38  		// 使用ftp代理服务器的主机、端口以及不需要使用ftp代理服务器的主机
39  		sysProperties.setProperty("ftp.proxyHost", "183.45.78.31");
40  		sysProperties.setProperty("ftp.proxyPort", "21");
41  		sysProperties.setProperty("ftp.nonProxyHosts", "localhost|192.168.0.*");
42  
43  		// socks代理服务器的地址与端口
44  		sysProperties.setProperty("socksProxyHost", "183.45.78.31");
45  		sysProperties.setProperty("socksProxyPort", "1080");
46  
47  		// 设置登陆到代理服务器的用户名和密码
48  		Authenticator.setDefault(new ProxyAuthenticator("userName", "Password"));
49  	}
50  
51  	public static void local() throws Exception {
52  		URL url = new URL("http://www.3lai8.com");
53  		// /创建代理服务器
54  		InetSocketAddress addr = new InetSocketAddress("192.168.0.254", 8080);
55  		// Proxy proxy = new Proxy(Proxy.Type.SOCKS, addr); // Socket 代理
56  		Proxy proxy = new Proxy(Proxy.Type.HTTP, addr); // http 代理
57  		Authenticator.setDefault(new ProxyAuthenticator("username", "password"));// 设置代理的用户和密码
58  		HttpURLConnection connection = (HttpURLConnection) url.openConnection(proxy);// 设置代理访问
59  		InputStreamReader in = new InputStreamReader(connection.getInputStream());
60  		BufferedReader reader = new BufferedReader(in);
61  		while (true) {
62  			String s = reader.readLine();
63  			if (s != null) {
64  				System.out.println(s);
65  			}
66  		}
67  	}
68  
69  }

71  class ProxyAuthenticator extends Authenticator {
72  	private String user;
73  	private String password;
74  
75  	public ProxyAuthenticator() {
76  		this.user = "";
77  		this.password = "";
78  	}
79  
80  	public ProxyAuthenticator(String user, String password) {
81  		this.user = user;
82  		this.password = password;
83  	}
84  
85  	public String getUser() {
86  		return user;
87  	}
88  
89  	public void setUser(String user) {
90  		this.user = user;
91  	}
92  
93  	public String getPassword() {
94  		return password;
95  	}
96  
97  	public void setPassword(String password) {
98  		this.password = password;
99  	}
