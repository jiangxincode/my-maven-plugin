11  public class CharByteConvert {
12  	public static byte[] getBytes(char[] chars) { // char转byte
13  		Charset cs = Charset.forName("UTF-8");
14  		CharBuffer cb = CharBuffer.allocate(chars.length);
15  		cb.put(chars);
16  		cb.flip();
17  		ByteBuffer bb = cs.encode(cb);
18  		return bb.array();
19  
20  	}
21  
22  	public static char[] getChars(byte[] bytes) { // byte转char
23  		Charset cs = Charset.forName("UTF-8");
24  		ByteBuffer bb = ByteBuffer.allocate(bytes.length);
25  		bb.put(bytes);
26  		bb.flip();
27  		CharBuffer cb = cs.decode(bb);
28  		return cb.array();
29  	}
30  	public static void main(String[] args) {
31  		byte[] origin = CharByteConvert.getBytes("This is a test!".toCharArray());
32  		char[] target = CharByteConvert.getChars(origin);
33  		System.out.println(target.toString());
34  	}
